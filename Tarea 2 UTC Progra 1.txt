ASIGNAR ACUMULADOR = 0
ASIGNAR k = 1
ASIGNAR PROM = 0
ASIGNAR MAX = 0

ESCRIBIR "CANTIDAD DE ESTUDIANTES"
LEER N

REPETIR DE k DE 1 A N
LEER NOTA

ACUMULADOR = ACUMULADOR + NOTA

SI NOTA > MAX
MAX = NOTA
FIN SI

TERMINO REPETIR

PROM = ACUMULADOR / N

IMPRIMIR "LA NOTA MAYOR ES: " MAX

ASIGNAR CONTADOR = 0
ASIGNAR k = 1

REPETIR DE k DE 1 A N
LEER NOTA

SI NOTA > PROM
CONTADOR = CONTADOR + 1
FIN SI

TERMINA REPETIR

IMPRIMIR "LOS ESTUDIANTES CON UNA NOTA SUPERIOR AL PROMEDIO SON: " CONTADOR
IMPRIMIR "LOS ESTUDIANTES CON UNA NOTA INFERIOR AL PROMEDIO SON: " (N-CONTADOR)
